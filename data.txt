#!/usr/bin/env python3
import json

raw_text = """Gas / Bloating: Feeling of fullness, stomach tightness, burping, passing gas, discomfort after meals. Often caused by overeating, spicy food, fizzy drinks, or indigestion. Mild cases may improve with light meals, walking, and avoiding trigger foods.

Neck Pain: Stiffness, pain while turning head, muscle tightness, headache. Often due to poor posture or sleeping position.

def parse_symptoms(text):
    symptoms = []
    for line in text.splitlines():
        line = line.strip()
        if not line:
            continue
        parts = line.split(":", 1)
        if len(parts) != 2:
            continue
        name = parts[0].strip()
        description = parts[1].strip()
        symptoms.append({"name": name, "description": description})
    return symptoms

def main():
    symptoms = parse_symptoms(raw_text)
    with open("symptoms.json", "w", encoding="utf-8") as f:
        json.dump(symptoms, f, indent=2, ensure_ascii=False)
    print(f"Wrote {len(symptoms)} symptom entries to symptoms.json")

if __name__ == "__main__":
    main()
